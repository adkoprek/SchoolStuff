from gturtle import *

def draw_hexagon(side):
    repeat 6:
        fd(side)
        rt(360 / 6)

## 3.3

```python
from gturtle import *

def draw_colored_triangle(side, color):
    setPenColor(color)
    repeat 3:
        fd(side)
        rt(360 / 3)

    fd(100)

makeTurtle()
lt(90)
draw_colored_triangle(100, 'green')
draw_colored_triangle(100, 'red')
draw_colored_triangle(100, 'blue')
```

## 3.5

```python
from gturtle import *

def leaf(height, width):
    lt(width / 2)
    repeat 2:
        repeat width:
            fd(height)
            right(1)
        rt(180 - width)
    rt(width / 2)

makeTurtle()
ht()
leaf(3, 100)
leaf(3, 150)
```

## 3.6

```python
from gturtle import *

def draw_stairs(width, height, number):
    repeat number:
        fd(height)
        rt(90)
        fd(width)
        lt(90)

makeTurtle()
draw_stairs(21, 16, 8)
```
