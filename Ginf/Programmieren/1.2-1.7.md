## 1.8

```python
import gturtle as gt


def draw_dashed_line(lenght):
    current_lenght = 0
    color = "red"
    while current_lenght < lenght:
        gt.setPenColor(color)
        if current_lenght + 5 > lenght:
            gt.fd(lenght - current_lenght)
        else:
            gt.fd(5)
        current_lenght += 5
        if color == "white":
            color = "red"
        else:
            color = "white"

def draw_shape(angle, side, number, dashed=False):
    repeat number:
        if dashed:
            draw_dashed_line(side)
        else:
            gt.fd(side)
        gt.rt(angle)


gt.makeTurtle()
gt.ht()
# draw_shape(33, 50, 11)
# draw_shape(24, 30, 15)
draw_shape(51.5, 100, 7, True)
```

## Mandelbrot

```python
import gturtle as gt


WIDTH = 512
HEIGHT = 512


def in_mandelbrot(max_iter, c):
    z = 0

    repeat max_iter:
        z = z ** 2 + c
        if abs(z) > 2:
            return False
    return True

def draw_bitmap(bitmap, width, height):
    gt.penUp()
    gt.fd(height / 2)
    gt.left(90)
    gt.fd(width / 2)
    gt.rt(180)
    gt.penDown()
    for y in range(height):
        for x in range(width):
            if bitmap[y][x] == False:
                gt.setPenColor("white")
            else:
                gt.setPenColor("black")
            gt.fd(1)

        gt.penUp()
        gt.rt(90)
        gt.fd(1)
        gt.rt(90)
        gt.fd(width)
        gt.rt(180)
        gt.penDown()

if __name__ == "__main__":     
    gt.makeTurtle()
    gt.ht()
    bitmap = []
    for y in range(HEIGHT):
        temp = []
        for x in range(WIDTH):
            c = 0.0075 * complex(x - WIDTH / 2, HEIGHT / 2 - y)
            if in_mandelbrot(20, c):
                temp.append(True)
            else:
                temp.append(False)
        bitmap.append(temp)


    draw_bitmap(bitmap, WIDTH, HEIGHT)
```

        gt.fd(side)
